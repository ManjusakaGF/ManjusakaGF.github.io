vue之flexible + px2rem

Aug 30 2018
安装淘宝适配库flexible

npm i lib-flexible --save

在你的main.js中引入flexible import 'lib-flexible/flexible'
安装px2remLoader

npm install px2rem-loader --save-dev

在build文件夹下的until.js添加配置来使用px2remLoader
在cssloaders配置下添加下列代码:
const px2remLoader = {
    loader: 'px2rem-loader',
    options: {
      remUnit: 75
    }
  }
在generate loader配置中添加px2remLoader
// generate loader string to be used with extract text plugin
function generateLoaders (loader, loaderOptions) {
  const loaders = options.usePostCSS ? [cssLoader, postcssLoader] : [cssLoader, px2remLoader]

  if (loader) {
    loaders.push({
      loader: loader + '-loader',
      options: Object.assign({}, loaderOptions, {
        sourceMap: options.sourceMap
      })
    })
  }

  // Extract CSS when that option is specified
  // (which is the case during production build)
  if (options.extract) {
    return ExtractTextPlugin.extract({
      use: loaders,
      fallback: 'vue-style-loader'
    })
  } else {
    return ['vue-style-loader'].concat(loaders)
  }
}
就可以自动转换px为rem了，最后附赠一条在组件内引入css文件的末尾一定要加 ;
<style>
@import '../static/css/reset.css';
</style>
不然 run npm dev的时候可是会报错哟~
